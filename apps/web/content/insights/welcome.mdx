import type { MDXComponents } from "mdx/types";

export function useMDXComponents(components: MDXComponents): MDXComponents {
  return {
    h1: (props) => (
      <h1
        className="mt-8 scroll-mt-24 text-4xl font-semibold tracking-tight text-kos-text"
        {...props}
      />
    ),
    h2: (props) => (
      <h2
        className="mt-10 scroll-mt-24 text-2xl font-semibold tracking-tight text-kos-text"
        {...props}
      />
    ),
    h3: (props) => (
      <h3
        className="mt-8 scroll-mt-24 text-xl font-semibold tracking-tight text-kos-text"
        {...props}
      />
    ),
    p: (props) => (
      <p className="mt-4 leading-7 text-kos-text/90" {...props} />
    ),
    a: (props) => (
      <a
        className="text-kos-gold underline underline-offset-4 hover:text-edge-green"
        {...props}
      />
    ),
    ul: (props) => <ul className="mt-4 list-disc pl-6" {...props} />,
    ol: (props) => <ol className="mt-4 list-decimal pl-6" {...props} />,
    li: (props) => <li className="mt-2 text-kos-text/90" {...props} />,
    blockquote: (props) => (
      <blockquote
        className="mt-6 border-l-2 border-kos-gold/40 pl-4 text-kos-text/80"
        {...props}
      />
    ),
    code: (props) => (
      <code
        className="rounded bg-kos-surface px-1.5 py-0.5 text-[0.95em] text-edge-green"
        {...props}
      />
    ),
    ...components,
  };
}
